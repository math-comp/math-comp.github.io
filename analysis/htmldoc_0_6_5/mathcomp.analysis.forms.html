                                    (s'_law : GRing.Scale.law s') (eqs' : s' = s'_law) :=
  ((∀ u', GRing.Linear.axiom (f^~ u') eqs)
  × (∀ u, GRing.Linear.axiom (f u) eqs'))%type.

   @axiom f s_law Ds s'_law Ds' → class_of f.
